Module Usage Examples

How to use the modular features of this portfolio as a library.

1. Import a single module

import { initTypewriter } from '../js/modules/typewriter.js';

// In your app:
document.addEventListener('DOMContentLoaded', initTypewriter);

2. Use the custom cursor on a new page

import { initCursor } from '../js/modules/cursor.js';

initCursor();
// Now you have a dot, ring, glow, trail, and click ripples.

3. Make any element magnetic

import { initMagnetic } from '../js/modules/magnetic.js';

document.querySelectorAll('.my-button').forEach(btn => {
  btn.classList.add('magnetic');
});
initMagnetic();

4. Add 3D tilt to a card

import { initTilt } from '../js/modules/tilt.js';

initTilt();
// All .project-card elements get the 3D tilt on hover.

5. Animate number counters

import { initCounters } from '../js/modules/counter.js';

// In your HTML:
// <div data-count="42">0</div>

initCounters();
// Counts up to 42 when scrolled into view.

6. Use the project data

import { projectData } from '../js/data.js';

const aiProject = projectData.ai;
console.log(aiProject.title);
// → "AGI Core — Neural Reasoning Engine"

7. Open a project modal programmatically

import { initModal } from '../js/modules/modal.js';

initModal();

// Trigger from anywhere:
document.querySelector('#openAI').addEventListener('click', () => {
  const modal = document.getElementById('modal');
  // populate with data, then:
  modal.classList.add('open');
});

8. Smooth scroll to anchor

import { initSmoothScroll } from '../js/modules/smoothScroll.js';

initSmoothScroll();
// All <a href="#section"> links now scroll smoothly.

9. Bootstrap everything (like main.js)

import { initLoader }       from './js/modules/loader.js';
import { initCursor }       from './js/modules/cursor.js';
import { initMagnetic }     from './js/modules/magnetic.js';
import { initTilt }         from './js/modules/tilt.js';
import { initTypewriter }   from './js/modules/typewriter.js';
import { initReveal }       from './js/modules/reveal.js';
import { initCounters }     from './js/modules/counter.js';
import { initActiveNav }    from './js/modules/nav.js';
import { initParallax }     from './js/modules/parallax.js';
import { initModal }        from './js/modules/modal.js';
import { initSmoothScroll } from './js/modules/smoothScroll.js';
import { initBackgrounds }  from './js/modules/background.js';
import { initHeroCanvas }   from './js/modules/heroCanvas.js';
import './js/modules/form.js';

document.addEventListener('DOMContentLoaded', () => {
  initLoader();
  initCursor();
  initMagnetic();
  initTilt();
  initTypewriter();
  initReveal();
  initCounters();
  initActiveNav();
  initParallax();
  initModal();
  initSmoothScroll();
  initBackgrounds();
  initHeroCanvas();
});

10. Use the design tokens

:root {
  --electric: #00f0ff;
  --purple:   #b400ff;
  --pink:     #ff00aa;
}

.my-element {
  background: var(--bg-1);
  color:      var(--text);
  border:     1px solid var(--border);
  box-shadow: 0 0 20px var(--electric);
}

11. Add a project to the data file

// js/data.js
export const projectData = {
  // ... existing projects
  newProject: {
    title: 'My New Project',
    icon: '✨',
    desc: 'Description here.',
    tags: ['Tag1', 'Tag2'],

  }
};
